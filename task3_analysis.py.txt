import pandas as pd
import os

# Load latest CSV
folder = "data"
files = os.listdir(folder)
csv_files = [f for f in files if f.endswith(".csv")]

latest_file = sorted(csv_files)[-1]
file_path = os.path.join(folder, latest_file)

df = pd.read_csv(file_path)

# Analysis
print("\n--- Analysis ---")

# Average score
print("Average Score:", df["score"].mean())

# Top category
print("\nTop Category:")
print(df["category"].value_counts())

# Most active authors
print("\nTop Authors:")
print(df["author"].value_counts().head(5))

# Highest scored post
top_post = df.loc[df["score"].idxmax()]
print("\nTop Post:")
print(top_post[["title", "score"]])
